 77def load_data_for_finetuning(
 78    config: DictConfig, processor: Callable | None = None
 79) -> IterableDatasetDict:
 80    """Load an audio dataset for finetuning.
 81
 82    Args:
 83        config:
 84            The Hydra configuration object.
 85        processor (optional):
 86            The processor to use for processing the audio and transcriptions. If `None`,
 87            then the processor is not used. Defaults to `None`.
 88
 89    Returns:
 90        The audio dataset.
 91
 92    Raises:
 93        ValueError:
 94            If the dataset is not supported.
 95    """
 96    # Note if we're on the main process, if we are running in a distributed setting
 97    is_main_process = os.getenv("RANK", "0") == "0"
 98
 99    all_datasets: list[IterableDataset] | list[Dataset] = list()
100    for dataset_name, dataset_config in config.datasets.items():
101        if is_main_process:
102            logger.info(f"Loading dataset {dataset_name!r}")
103
104        # Load from disk if the dataset ID is a path and it is stored as an arrow dataset
105        if Path(dataset_config.id).exists():
106            train_path = Path(dataset_config.id) / dataset_config.train_name
107            data_files = list(map(str, train_path.glob("data-*.arrow")))
108            if len(data_files) == 0:
109                try:
110                    ds = load_dataset(
111                        path=dataset_config.id,
112                        name=dataset_config.subset,
113                        split=dataset_config.train_name,
114                        streaming=config.streaming,
115                        cache_dir=config.cache_dir,
116                    )
117
118                # In case a single split has been stored to disk, we load it directly
119                except ValueError as e:
120                    if "load_from_disk" not in str(e):
121                        raise e
122                    ds = Dataset.load_from_disk(dataset_path=dataset_config.id)
123            else:
124                try:
125                    ds = load_dataset(
126                        "arrow",
127                        data_files=data_files,
128                        split=dataset_config.train_name,
129                        streaming=config.streaming,
130                        cache_dir=config.cache_dir,
131                    )
132                except ValueError:
133                    ds = load_dataset(
134                        "arrow",
135                        data_files=data_files,
136                        split="train",
137                        streaming=config.streaming,
138                        cache_dir=config.cache_dir,
139                    )
140
141        # Load dataset from the Hugging Face Hub. The HUGGINGFACE_HUB_TOKEN is only
142        # used during CI - normally it is expected that the user is logged in to the
143        # Hugging Face Hub using the `huggingface-cli login` command.
144        else:
145            ds = load_dataset(
146                path=dataset_config.id,
147                name=dataset_config.subset,
148                split=dataset_config.train_name,
149                token=os.getenv("HUGGINGFACE_HUB_TOKEN", True),
150                streaming=config.streaming,
151                trust_remote_code=True,
152                cache_dir=config.cache_dir,
153            )
154
155        assert isinstance(
156            ds, Dataset | IterableDataset
157        ), f"Unsupported dataset type: {type(ds)}"
158
159        if dataset_config.text_column != "text":
160            ds = ds.rename_column(dataset_config.text_column, "text")
161        if dataset_config.audio_column != "audio":
162            ds = ds.rename_column(dataset_config.audio_column, "audio")
163
164        if dataset_config.filter_dataset:
165            ds = filter_dataset(
166                dataset=ds,
167                audio_column="audio",
168                min_seconds_per_example=config.min_seconds_per_example,
169                max_seconds_per_example=config.max_seconds_per_example,
170                is_main_process=is_main_process,
171                num_proc=config.dataset_num_workers,
172            )
173
174        if dataset_config.process_dataset:
175            ds = ds.remove_columns(
176                column_names=[
177                    column
178                    for column in ds.column_names or list()
179                    if column not in ["audio", "text"]
180                ]
181            ).shuffle(seed=config.seed)
182            ds = process_dataset(
183                dataset=ds,
184                clean_text=config.model.clean_text,
185                lower_case=config.model.lower_case,
186                characters_to_keep=config.characters_to_keep,
187                text_column="text",
188                audio_column="audio",
189                convert_numerals=False,
190                remove_input_dataset_columns=True,
191                cast_to_sampling_rate=config.model.sampling_rate,
192                processor=processor,
193                num_proc=config.dataset_num_workers,
194            )
195
196        all_datasets.append(ds)
197
198    assert len(all_datasets) > 0, "No datasets were loaded"
199
200    if len(all_datasets) > 1:
201        if is_main_process:
202            logger.info("Interleaving datasets")
203            if config.dataset_probabilities is None and len(all_datasets) > 1:
204                logger.warning(
205                    "No dataset probabilities were specified for the training split. "
206                    "This means that each dataset will be sampled with equal "
207                    "probability, which means that the smaller datasets will be "
208                    "sampled more often than the larger datasets. This is probably "
209                    "not what you want."
210                )
211
212        probabilities = config.dataset_probabilities
213        if probabilities is None:
214            probabilities = [1 / len(all_datasets)] * len(all_datasets)
215            probabilities[-1] = 1 - sum(probabilities[:-1])
216        elif sum(probabilities) != 1:
217            raise ValueError(
218                f"Dataset probabilities must sum to 1, but sum to {sum(probabilities)}"
219            )
220
221        train = interleave_datasets(
222            datasets=[ds for ds in all_datasets],
223            probabilities=probabilities,
224            seed=config.seed,
225            split=NamedSplit("train"),
226            stopping_strategy="all_exhausted",
227        )
228    else:
229        train = all_datasets[0]
230
231    data_dict = dict(train=train)
232    dataset = IterableDatasetDict(data_dict)
233
234    if is_main_process:
235        logger.info("Loading CoRal validation dataset...")
236
237    val = load_dataset(
238        path=config.evaluation_dataset.id,
239        name=config.evaluation_dataset.subset,
240        split=config.evaluation_dataset.val_name,
241        token=os.getenv("HUGGINGFACE_HUB_TOKEN", True),
242        streaming=True,
243        trust_remote_code=True,
244        cache_dir=config.cache_dir,
245    )
246    assert isinstance(val, IterableDataset)
247    if not config.streaming:
248        val = convert_iterable_dataset_to_dataset(
249            iterable_dataset=val,
250            split_name="val",
251            dataset_id=config.evaluation_dataset.id.replace("/", "--") + "-validation",
252            cache_dir=config.cache_dir,
253        )
254    if config.evaluation_dataset.text_column != "text":
255        val = val.rename_column(config.evaluation_dataset.text_column, "text")
256    if config.evaluation_dataset.audio_column != "audio":
257        val = val.rename_column(config.evaluation_dataset.audio_column, "audio")
258
259    val = process_dataset(
260        dataset=val,
261        clean_text=config.model.clean_text,
262        lower_case=config.model.lower_case,
263        characters_to_keep=config.characters_to_keep,
264        text_column="text",
265        audio_column="audio",
266        convert_numerals=False,
267        remove_input_dataset_columns=True,
268        cast_to_sampling_rate=config.model.sampling_rate,
269        processor=processor,
270        num_proc=config.dataset_num_workers,
271    )
272    dataset["val"] = val
273
274    return dataset

277def load_dataset_for_evaluation(config: DictConfig) -> Dataset:
278    """Load the evaluation dataset.
279
280    Args:
281        config:
282            The Hydra configuration object.
283
284    Returns:
285        A DatasetDict containing the validation and test datasets.

288    is_main_process = os.getenv("RANK", "0") == "0"

313        trust_remote_code=True,

339        cast_to_sampling_rate=config.sampling_rate,
340        convert_numerals=True,
341    )
342
343    if config.cache_dir:
344        dataset.save_to_disk(dataset_path=eval_dataset_path)

349def filter_dataset(
350    dataset: Data,
351    audio_column: str,
352    min_seconds_per_example: int,
353    max_seconds_per_example: int,
354    is_main_process: bool,
355    num_proc: int | None = None,
356) -> Data:
357    """Filter the dataset.
358
359    Note that this removes samples from the dataset.
360
361    Args:
362        dataset:
363            The dataset to filter.
364        audio_column:
365            The name of the column containing the audio.
366        min_seconds_per_example:
367            The minimum number of seconds that an example can have.
368        max_seconds_per_example:
369            The maximum number of seconds that an example can have.
370        is_main_process:
371            Whether the current process is the main process.
372        num_proc (optional):
373            The number of processes to use for filtering the dataset. If `None`, then
374            no multiprocessing is used. Defaults to `None`.
375
376    Returns:
377        The filtered dataset.
378    """
379    num_samples_before = len(dataset) if isinstance(dataset, Sized) else 0
380
381    filter_fn = partial(
382        filter_example,
383        audio_column=audio_column,
384        min_seconds_per_example=min_seconds_per_example,
385        max_seconds_per_example=max_seconds_per_example,
386    )
387    if isinstance(dataset, Dataset | DatasetDict):
388        filtered = dataset.filter(
389            function=filter_fn, num_proc=num_proc, desc="Filtering dataset"
390        )
391    else:
392        filtered = dataset.filter(function=filter_fn)
393
394    # Add info back in the filtered dataset, as it gets removed after calling `filter`
395    if isinstance(dataset, Dataset | IterableDataset):
396        filtered.info.features = dataset.info.features
397    else:
398        for split_name in dataset.keys():
399            dataset[split_name].info.features = filtered[split_name].info.features
400
401    if isinstance(dataset, Sized) and is_main_process:
402        num_samples_removed = num_samples_before - len(dataset)
403        logger.info(f"Removed {num_samples_removed:,} samples from the dataset")
404
405    return filtered

408def filter_example(
409    sample: dict[str, Any],
410    audio_column: str,
411    min_seconds_per_example: int,
412    max_seconds_per_example: int,
413) -> bool:
414    """Filter samples based on the validation status.
415
416    Args:
417        sample:
418            The sample to filter.
419        audio_column:
420            The name of the column containing the audio.
421        min_seconds_per_example:
422            The minimum number of seconds that an example can have.
423        max_seconds_per_example:
424            The maximum number of seconds that an example can
425
426    Returns:
427        Whether the sample should be kept.
428    """
429    # Filtering based on audio
430    audio = sample[audio_column]
431    if audio["array"].shape[0] <= audio["sampling_rate"] * min_seconds_per_example:
432        return False
433    if audio["array"].shape[0] >= audio["sampling_rate"] * max_seconds_per_example:
434        return False
435
436    # Filtering based on validation
437    if "validated" in sample and sample["validated"] == "rejected":
438        return False
439
440    return True

443def process_dataset(
444    dataset: Data,
445    clean_text: bool,
446    lower_case: bool,
447    characters_to_keep: Iterable[str] | None,
448    text_column: str,
449    remove_input_dataset_columns: bool,
450    audio_column: str | None,
451    convert_numerals: bool,
452    num_proc: int | None = None,
453    cast_to_sampling_rate: int | None = None,
454    processor: Callable | None = None,
455) -> Data:
456    """Process the dataset.
457
458    Note that this does not remove any samples from the dataset.
459
460    Args:
461        dataset:
462            The dataset to be cleaned.
463        clean_text:
464            Whether to clean the text.
465        lower_case:
466            Whether to make the text lower case. Only relevant if `clean_text` is True.
467        characters_to_keep:
468            All the characters that should be kept in the transcriptions. Can be None if
469            all characters should be kept. Only relevant if `clean_text` is True.
470        text_column:
471            The name of the column containing the text.
472        remove_input_dataset_columns:
473            Whether to remove all input dataset columns from the output dataset.
474        audio_column:
475            The name of the column containing the audio. Can be `None` if the dataset
476            does not have an audio column.
477        convert_numerals:
478            Whether to convert numerals to words.
479        num_proc (optional):
480            The number of processes to use for processing the dataset. If `None`, then
481            no multiprocessing is used. Defaults to `None`.
482        cast_to_sampling_rate (optional):
483            The sampling rate to cast the audio to. If `None`, then the audio is not
484            cast. Defaults to `None`.
485        processor (optional):
486            The processor to use for processing the audio and transcriptions. If `None`,
487            then the processor is not used. Defaults to `None`.
488
489    Returns:
490        The cleaned dataset.
491    """
492    if audio_column is not None:
493        dataset = dataset.cast_column(
494            column=audio_column, feature=Audio(sampling_rate=cast_to_sampling_rate)
495        )
496
497    if isinstance(dataset, Dataset) or isinstance(dataset, IterableDataset):
498        column_names = dataset.column_names
499    elif isinstance(dataset, DatasetDict) or isinstance(dataset, IterableDatasetDict):
500        column_names = dataset["train"].column_names
501
502    map_fn = partial(
503        process_example,
504        characters_to_keep=characters_to_keep,
505        conversion_dict=DEFAULT_CONVERSION_DICT,
506        text_column=text_column,
507        audio_column=audio_column,
508        clean_text=clean_text,
509        lower_case=lower_case,
510        convert_numerals=convert_numerals,
511        processor=processor,
512    )
513    if isinstance(dataset, Dataset | DatasetDict):
514        mapped = dataset.map(
515            function=map_fn,
516            num_proc=num_proc,
517            desc="Processing dataset",
518            remove_columns=column_names if remove_input_dataset_columns else None,
519        )
520    else:
521        mapped = dataset.map(function=map_fn, remove_columns=column_names)
522
523    return mapped

526def process_example(
527    example: dict,
528    characters_to_keep: Iterable[str] | None,
529    conversion_dict: dict[str, str],
530    text_column: str,
531    audio_column: str | None,
532    clean_text: bool,
533    lower_case: bool,
534    convert_numerals: bool,
535    processor: Callable | None,
536) -> dict:
537    """Helper function which cleans a single example.
538
539    Args:
540        example:
541            The example to be cleaned.
542        characters_to_keep:
543            All the characters that should be kept in the transcriptions. Can be None if
544            all characters should be kept.
545        conversion_dict:
546            A dictionary of characters to be converted.
547        text_column:
548            The name of the column containing the text.
549        audio_column:
550            The name of the column containing the audio. Can be `None` if the dataset
551            does not have an audio column.
552        clean_text:
553            Whether to clean the text.
554        lower_case:
555            Whether to make the text lower case.
556        convert_numerals:
557            Whether to convert numerals to words.
558        processor:
559            The processor to use for processing the audio and transcriptions. If `None`,
560            then the processor is not used. Requires `audio_column` to be specified.
561
562    Returns:
563        The cleaned example.
564    """
565    doc = example[text_column]
566
567    if convert_numerals and re.search(pattern=NUMERAL_REGEX, string=doc):
568        doc = "".join(
569            convert_numeral_to_words(numeral=maybe_numeral)
570            for maybe_numeral in re.split(pattern=NUMERAL_REGEX, string=doc)
571            if maybe_numeral is not None
572        )
573
574    if lower_case:
575        doc = doc.lower()
576
577    # Normalise the transcription, which uniformises the characters. For instance, the
578    # "long dash" (－) is converted to the normal dash (-).
579    if clean_text:
580        doc = normalize("NFKC", doc)
581
582        for key, value in conversion_dict.items():
583            doc = doc.replace(key, value)
584
585        # Remove all non-standard characters
586        if characters_to_keep is not None:
587            characters_to_keep = "".join(char for char in characters_to_keep)
588            if lower_case:
589                characters_to_keep = characters_to_keep.lower()
590            else:
591                characters_to_keep = (
592                    characters_to_keep.upper() + characters_to_keep.lower()
593                )
594            non_standard_characters_regex = re.compile(
595                f"[^{re.escape(characters_to_keep + ' |')}]"
596            )
597            doc = re.sub(non_standard_characters_regex, " ", doc.strip())
598
599        # Replace superfluous spaces
600        doc = re.sub(r" +", " ", doc)
601
602        # Strip each newline
603        doc = "\n".join([line.strip() for line in doc.split("\n")]).strip("\n")
604
605    # Re-assign the cleaned transcription
606    example[text_column] = doc
607
608    if processor is None:
609        return example
610
611    # Prepare audio
612    audio = example[audio_column]
613    sampling_rate = audio["sampling_rate"]
614    processed = processor(audio["array"], sampling_rate=sampling_rate)
615    if "input_values" in processed:
616        example["input_values"] = processed.input_values[0]
617        example["num_seconds"] = len(example["input_values"]) / sampling_rate
618    if "input_features" in processed:
619        example["input_features"] = processed.input_features[0]
620        example["num_seconds"] = len(example["input_features"]) / sampling_rate
621
622    # Prepare transcriptions
623    example["labels"] = processor(text=example[text_column], truncation=True).input_ids
624    example["input_length"] = len(example["labels"])
625
626    return example

528    example: dict,
529    characters_to_keep: Iterable[str] | None,
530    conversion_dict: dict[str, str],
531    text_column: str,
532    audio_column: str | None,

535    convert_numerals: bool,

538    """Helper function which cleans a single example.
539
540    Args:
541        example:
542            The example to be cleaned.
543        characters_to_keep:
544            All the characters that should be kept in the transcriptions. Can be None if
545            all characters should be kept.
546        conversion_dict:
547            A dictionary of characters to be converted.
548        text_column:
549            The name of the column containing the text.
550        audio_column:
551            The name of the column containing the audio. Can be `None` if the dataset
552            does not have an audio column.
553        clean_text:
554            Whether to clean the text.
555        lower_case:
556            Whether to make the text lower case.
557        convert_numerals:
558            Whether to convert numerals to words.

561            then the processor is not used. Requires `audio_column` to be specified.
562
563    Returns:
564        The cleaned example.
565    """
566    doc = example[text_column]
567
568    if convert_numerals and re.search(pattern=NUMERAL_REGEX, string=doc):
569        doc = "".join(
570            convert_numeral_to_words(numeral=maybe_numeral)
571            for maybe_numeral in re.split(pattern=NUMERAL_REGEX, string=doc)
572            if maybe_numeral is not None
573        )
574
575    if lower_case:
576        doc = doc.lower()
577
578    # Normalise the transcription, which uniformises the characters. For instance, the
579    # "long dash" (－) is converted to the normal dash (-).
580    if clean_text:
581        doc = normalize("NFKC", doc)
582
583        for key, value in conversion_dict.items():
584            doc = doc.replace(key, value)
585
586        # Remove all non-standard characters
587        if characters_to_keep is not None:
588            characters_to_keep = "".join(char for char in characters_to_keep)
589            if lower_case:
590                characters_to_keep = characters_to_keep.lower()
591            else:
592                characters_to_keep = (
593                    characters_to_keep.upper() + characters_to_keep.lower()
594                )
595            non_standard_characters_regex = re.compile(
596                f"[^{re.escape(characters_to_keep + ' |')}]"
597            )
598            doc = re.sub(non_standard_characters_regex, " ", doc.strip())
599
600        # Replace superfluous spaces
601        doc = re.sub(r" +", " ", doc)
602
603        # Strip each newline
604        doc = "\n".join([line.strip() for line in doc.split("\n")]).strip("\n")
605
606    # Re-assign the cleaned transcription
607    example[text_column] = doc
608
609    if processor is None:
610        return example
611
612    # Prepare audio
613    audio = example[audio_column]
614    sampling_rate = audio["sampling_rate"]
615    processed = processor(audio["array"], sampling_rate=sampling_rate)
616    if "input_values" in processed:
617        example["input_values"] = processed.input_values[0]
618        example["num_seconds"] = len(example["input_values"]) / sampling_rate
619    if "input_features" in processed:
620        example["input_features"] = processed.input_features[0]
621        example["num_seconds"] = len(example["input_features"]) / sampling_rate
622
623    # Prepare transcriptions
624    example["labels"] = processor(text=example[text_column], truncation=True).input_ids
625    example["input_length"] = len(example["labels"])
